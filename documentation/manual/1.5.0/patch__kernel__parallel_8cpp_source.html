51 void PatchKernel::setCommunicator(MPI_Comm communicator)
52 {
53  // Communication can be set just once
54  if (isCommunicatorSet()) {
55  throw std::runtime_error ("Patch communicator can be set just once");
56  }
57 
58  // The communicator has to be valid
59  if (communicator == MPI_COMM_NULL) {
60  throw std::runtime_error ("Patch communicator is not valid");
61  }
62 
63  // Creat a copy of the user-specified communicator
64  //
65  // No library routine should use MPI_COMM_WORLD as the communicator;
66  // instead, a duplicate of a user-specified communicator should always
67  // be used.
68  MPI_Comm_dup(communicator, &m_communicator);
69 
70  // Get MPI information
71  MPI_Comm_size(m_communicator, &m_nProcessors);
72  MPI_Comm_rank(m_communicator, &m_rank);
73 
74  // Set parallel data for the VTK output
75  if (m_nProcessors > 1) {
76  m_vtk.setParallel(m_nProcessors, m_rank);
77  }
78 }

349 std::vector<adaption::Info> PatchKernel::partitioningPrepare(const std::vector<int> &cellRanks, bool trackPartitioning)
350 {
351  std::vector<adaption::Info> partitioningData;
352 
353  // Communicator has to be set
354  if (!isCommunicatorSet()) {
355  throw std::runtime_error ("There is no communicator set for the patch.");
356  }
357 
358  // Check partitioning status
359  PartitioningStatus partitioningStatus = getPartitioningStatus(true);
360  if (partitioningStatus == PARTITIONING_UNSUPPORTED) {
361  return partitioningData;
362  } else if (partitioningStatus != PARTITIONING_CLEAN) {
363  throw std::runtime_error ("A partitioning is already in progress.");
364  }
365 
366  // Build the list of ids to be sent
367  auto cellItr = cellBegin();
368  for (int k = 0; k < getInternalCount(); ++k) {
369  const int &rank = cellRanks[k];
370  if (rank == getRank()) {
371  cellItr++;
372  continue;
373  }
374 
375  m_partitioningSends[rank].push_back(cellItr->getId());
376 
377  cellItr++;
378  }
379 
380  // Local sender-receiver pairs
381  std::vector<int> localPairs;
382  localPairs.reserve(2 * m_partitioningSends.size());
383  for (const auto &entry : m_partitioningSends) {
384  localPairs.push_back(getRank());
385  localPairs.push_back(entry.first);
386  }
387 
388  // Prepare the communication for exchanging the sender/receiver pairs
389  int globalPairsSize = localPairs.size();
390  std::vector<int> globalPairsSizes(getProcessorCount());
391  MPI_Allgather(&globalPairsSize, 1, MPI_INT, globalPairsSizes.data(), 1, MPI_INT, getCommunicator());
392 
393  std::vector<int> globalPairsOffsets(getProcessorCount());
394  globalPairsOffsets[0] = 0;
395  for (int i = 1; i < getProcessorCount(); ++i) {
396  globalPairsOffsets[i] = globalPairsOffsets[i-1] + globalPairsSizes[i-1];
397  }
398 
399  // Global sender-receiver pairs
400  m_partitioningPairs.resize(globalPairsOffsets.back() + globalPairsSizes.back());
401 
402  MPI_Allgatherv(localPairs.data(), localPairs.size(), MPI_INT, m_partitioningPairs.data(),
403  globalPairsSizes.data(), globalPairsOffsets.data(), MPI_INT,
404  getCommunicator());
405 
406  // Build the information on the cells that will be sent
407  if (trackPartitioning) {
408  for (const auto &entry : m_partitioningSends) {
409  int receiver = entry.first;
410  const std::vector<long> &ids = entry.second;
411 
412  partitioningData.emplace_back();
413  adaption::Info &partitioningInfo = partitioningData.back();
414  partitioningInfo.entity = adaption::ENTITY_CELL;
415  partitioningInfo.type = adaption::TYPE_PARTITION_SEND;
416  partitioningInfo.rank = receiver;
417  partitioningInfo.previous = ids;
418  }
419  }
420 
421  return partitioningData;
422 }

490 std::vector<adaption::Info> PatchKernel::partitioningAlter(bool trackPartitioning, bool squeezeStorage)
491 {
492  std::vector<adaption::Info> partitioningData;
493 
494  // Check partitioning status
495  PartitioningStatus partitioningStatus = getPartitioningStatus();
496  if (partitioningStatus == PARTITIONING_UNSUPPORTED || partitioningStatus == PARTITIONING_CLEAN) {
497  return partitioningData;
498  } else if (partitioningStatus != PARTITIONING_PREPARED) {
499  throw std::runtime_error ("The prepare function has no been called.");
500  }
501 
502  // Begin patch alteration
503  beginAlteration();
504 
505  // Alter patch
506  if (m_partitioningSends.empty()) {
507  partitioningData = _partitioningAlter(trackPartitioning);
508  } else {
509  std::vector<long> emptyCellList;
510  for (size_t i = 0; i < m_partitioningPairs.size(); i += 2) {
511  int sender = m_partitioningPairs[i];
512  int receiver = m_partitioningPairs[i+1];
513 
514  std::vector<long> *ids;
515  if (sender == getRank()) {
516  ids = &(m_partitioningSends[receiver]);
517  } else {
518  ids = &emptyCellList;
519  }
520 
521  adaption::Info partitioningInfo = sendCells(sender, receiver, *ids);
522  if (trackPartitioning && partitioningInfo.type != adaption::TYPE_NONE) {
523  partitioningData.push_back(std::move(partitioningInfo));
524  }
525  }
526  }
527 
528  // End patch alteration
529  endAlteration(squeezeStorage);
530 
531  // The patch is now partitioned
532  setPartitioned(true);
533 
534  // Update the status
535  setPartitioningStatus(PARTITIONING_ALTERED);
536 
537  return partitioningData;
538 }

546 void PatchKernel::partitioningCleanup()
547 {
548  PartitioningStatus partitioningStatus = getPartitioningStatus();
549  if (partitioningStatus == PARTITIONING_UNSUPPORTED || partitioningStatus == PARTITIONING_CLEAN) {
550  return;
551  } else if (partitioningStatus == PARTITIONING_PREPARED) {
552  throw std::runtime_error ("It is not yet possible to abort a partitioning.");
553  } else if (partitioningStatus != PARTITIONING_ALTERED) {
554  throw std::runtime_error ("The alter function has no been called.");
555  }
556 
557  // Clean-up the partitioning
558  _partitioningCleanup();
559 
560  if (m_partitioningSends.empty()) {
561  std::unordered_map<int, std::vector<long>>().swap(m_partitioningSends);
562  std::vector<int>().swap(m_partitioningPairs);
563  }
564 
565  // Update the status
566  setPartitioningStatus(PARTITIONING_CLEAN);
567 }

623 double PatchKernel::evalPartitioningUnbalance()
624 {
625  if (!isPartitioned()) {
626  return 0.;
627  }
628 
629  // Evaluate partition weight
630  double localWeight = getInternalCount();
631 
632  // Evalaute global weights
633  double totalWeight;
634  MPI_Allreduce(&localWeight, &totalWeight, 1, MPI_DOUBLE, MPI_SUM, getCommunicator());
635 
636  double minimumWeight;
637  MPI_Allreduce(&localWeight, &minimumWeight, 1, MPI_DOUBLE, MPI_MIN, getCommunicator());
638 
639  double maximumWeight;
640  MPI_Allreduce(&localWeight, &maximumWeight, 1, MPI_DOUBLE, MPI_MAX, getCommunicator());
641 
642  // Evaluate the unbalance
643  double unbalance = (maximumWeight - minimumWeight) / totalWeight;
644 
645  return unbalance;
646 }

950 void PatchKernel::addGhostsToExchangeInfo(const std::vector<long> &ghostIds)
951 {
952  // Add the ghost to the targets
953  std::unordered_set<int> ranks;
954  for (const long ghostId : ghostIds) {
955  // Rank of the ghost
956  int rank = m_ghostOwners[ghostId];
957  ranks.insert(rank);
958 
959  // Add the ghost to the targets

965  std::vector<long> &rankTargets = m_ghostExchangeTargets[rank];
966  rankTargets.erase(std::unique(rankTargets.begin(), rankTargets.end()), rankTargets.end());
967  std::sort(rankTargets.begin(), rankTargets.end(), CellPositionLess(*this));
968  }
969 
970  // Build the sources
971  for (const int rank : ranks) {
972  buildGhostExchangeSources(rank);

1062 std::vector<long> PatchKernel::_findGhostExchangeSources(int rank)
1063 {
1064  // The internal neighbours of the ghosts will be sources for the rank
1065  std::vector<long> neighIds;
1066  std::unordered_set<long> exchangeSources;
1067  exchangeSources.reserve(m_ghostExchangeTargets[rank].size());
1068  for (long ghostId : m_ghostExchangeTargets[rank]) {
1069  neighIds.clear();
1070  findCellNeighs(ghostId, &neighIds);
1071  for (long neighId : neighIds) {
1072  if (m_ghostOwners.count(neighId) > 0) {
1073  continue;
1074  }
1075 
1076  exchangeSources.insert(neighId);
1077  }
1078  }
1079 
1080  return std::vector<long>(exchangeSources.begin(), exchangeSources.end());
1081 }

1123  adaptionInfo.rank = recvRank;

1129  // Later, a layer of surrounding cells will be added.

1141  }
1142

1144  cellRankOnReceiver.insert({{cellId, recvRank}});

1149  //
1150  // Processors that have, among their ghost, cells that will be sent to
1151  // the receiver need to be notified about the ownership change. We will
1152  // communicate to the neighbours only the index in the exchange info
1153  // structure of the cells that have change ownership.

1155  // We need to create the notification now that the set with the cells

1173

1216  } else {

1218  if (ownerRank == recvRank) {
1219  continue;
1220  }
1221  }
1222

1224  cellRankOnReceiver.insert({{neighId, ownerRank}});
1225  }
1226  }

1229  // the receiver (these cells are ghost owned by this processor). Mark
1230  // is as such to help the receiver to remove duplicate cells.

1236  long cellId = exchangeSources[k];

1238  senderGhostsToPromote.insert({{cellId, k}});
1239  }
1240  }
1241  }

1245  //

1277  vertexBuffer << m_vertices[vertexId];
1278  }

1302  // Index of the cell in the exchange info structure
1303  long index = entry.second;
1304  cellBuffer << index;
1305 
1306  // Id of the cell on this processor
1307  long cellId = entry.first;
1308  cellBuffer << cellId;
1309  }

1312  cellBufferSize += sizeof(long);
1313  for (const long &cellId : cellsToCommunicate) {
1314  cellBufferSize += sizeof(int) + m_cells[cellId].getBinarySize();
1315  }
1316  cellBuffer.setSize(cellBufferSize);
1317

1323

1325  cellBuffer << m_cells[cellId];
1326  }

1329  throw std::runtime_error ("Cell buffer size does not match calculated size");
1330  }
1331 
1332  // Communication
1333  MPI_Send(&cellBufferSize, 1, MPI_LONG, recvRank, 20, m_communicator);
1334  MPI_Send(cellBuffer.data(), cellBuffer.getSize(), MPI_CHAR, recvRank, 21, m_communicator);

1337  // Send ownership notifications
1338  //
1339  for (const auto &ownershipNotification : ownershipNotifications) {
1340  int neighRank = ownershipNotification.first;
1341  const auto &notificationList = ownershipNotification.second;
1342 
1343  // Communicate the size of the notification

1358

1364  // Update cells that no longer belong to this processor
1365  //
1366 
1367  // Add the sned ids int the adaption info
1368  //
1369  // The ids will be sorted by the position of the cells, this is the
1370  // same order that will be used on the processor that has received the
1371  // octants. Since the order is the same, the two processors are able
1372  // to exchange cell data without any additional extra communication
1373  // (they already know the list of cells for which data is needed and
1374  // the order in which these data will be sent).
1375  for (long cellId : cellsToSend) {
1376  adaptionInfo.previous.push_back(cellId);
1377  }
1378 
1379  std::sort(adaptionInfo.previous.begin(), adaptionInfo.previous.end(), CellPositionLess(*this));
1380 
1381  // Delete sent cells or mark them as ghosts owned by the receiver.
1382  for (long cellId : cellsToSend) {
1383  // Check if a cell has to be delete or is a ghost owned by the receiver
1384  //
1385  // A cell will become a ghost if at least one of his neighbours is
1386  // an internal cell. If the processors is sending all its cells
1387  // there will be no ghosts cell.
1388  bool moveToGhosts = false;
1389  if (cellsToSendFrame.count(cellId) == 0) {
1390  moveToGhosts = false;
1391  } else {
1392  neighIds.clear();
1393  findCellNeighs(cellId, &neighIds);
1394  for (long neighId : neighIds) {
1395  if (m_ghostOwners.count(neighId) == 0) {
1396  moveToGhosts = true;
1397  break;
1398  }
1399  }
1400  }
1401 
1402  // Delete the cell or mark is as a ghost owned by the receiver.
1403  if (moveToGhosts) {
1404  moveInternal2Ghost(cellId);
1405  setGhostOwner(cellId, recvRank, false);
1406  } else {
1407  unsetGhostOwner(cellId, false);
1408  deleteCell(cellId, true, true);
1409  }
1410  }
1411 
1412  // Delete stale ghosts
1413  //
1414  // Loop over all the ghosts and keep only the cells that have at least
1415  // one internal neighbour.
1416  std::unordered_set<int> involvedRanks;
1417  involvedRanks.insert(recvRank);
1418  for (const auto &entry : ownershipNotifications) {
1419  involvedRanks.insert(entry.first);
1420  }
1421 
1422  auto itr = m_ghostOwners.cbegin();
1423  while (itr != m_ghostOwners.cend()) {
1424  long ghostId = itr->first;
1425 
1426  neighIds.clear();
1427  findCellNeighs(ghostId, &neighIds);
1428  bool keep = false;
1429  for (long neighId : neighIds) {
1430  if (m_ghostOwners.count(neighId) == 0) {
1431  keep = true;
1432  break;
1433  }
1434  }
1435 
1436  auto nextItr = itr;
1437  nextItr++;
1438  if (!keep) {
1439  long ghostOwner = itr->second;
1440  involvedRanks.insert(ghostOwner);
1441 
1442  unsetGhostOwner(ghostId, false);
1443  deleteCell(ghostId, true, true);
1444  }
1445  itr = nextItr;
1446  }
1447 
1448  m_cells.flush();
1449 
1450  // Delete orphan vertices
1451  deleteOrphanVertices();
1452 
1453  // Rebuild ghost information
1454  std::vector<int> involvedRankList(involvedRanks.begin(), involvedRanks.end());
1455  buildGhostExchangeInfo(involvedRankList);
1456 
1457  // Return adaption info
1458  return adaptionInfo;
1459 }
