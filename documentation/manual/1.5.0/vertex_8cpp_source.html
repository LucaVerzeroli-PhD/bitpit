254 void Vertex::scale(const std::array<double, 3> &scaling, const std::array<double, 3> &center)
255 {
256  m_coords[0] = center[0] + scaling[0] * (m_coords[0] - center[0]);
257  m_coords[1] = center[1] + scaling[1] * (m_coords[1] - center[1]);
258  m_coords[2] = center[2] + scaling[2] * (m_coords[2] - center[2]);
259 }
