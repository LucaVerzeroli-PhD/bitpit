158 std::array<double, 3> VolUnstructured::evalInterfaceNormal(const long &id) const
159 {
160  std::array<std::array<double, 3>, ReferenceElementInfo::MAX_ELEM_VERTICES> coordinatesStaticPool;
161 
162  const Interface &interface = getInterface(id);
163  ConstProxyVector<std::array<double, 3>> vertexCoordinates = getElementVertexCoordinates(interface, coordinatesStaticPool.data());
164 
165  std::array<double, 3> orientation = {{0., 0., 0.}};
166  if (!isThreeDimensional()) {
167  long ownerId = interface.getOwner();
168  const Cell &owner = getCell(ownerId);
169 
170  ConstProxyVector<long> ownerVertexIds = owner.getVertexIds();
171  const int nOwnerVertices = ownerVertexIds.size();
172 
173  const std::array<double, 3> &V_A = getVertex(ownerVertexIds[0]).getCoords();
174  const std::array<double, 3> &V_B = getVertex(ownerVertexIds[1]).getCoords();
175  const std::array<double, 3> &V_Z = getVertex(ownerVertexIds[nOwnerVertices - 1]).getCoords();
176 
177  orientation = crossProduct(V_B - V_A, V_Z - V_A);
178  }
179 
180  return interface.evalNormal(vertexCoordinates.data(), orientation);
181 }
