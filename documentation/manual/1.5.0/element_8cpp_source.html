166 void Element::Tesselation::importPolygon(const std::vector<int> &vertexIds)
167 {
168  int nVertices = vertexIds.size();
169  if (nVertices == 3 || nVertices == 4) {
170  m_nTiles++;
171  m_types.push_back(nVertices == 3 ? ElementType::TRIANGLE : ElementType::QUAD);
172  m_connects.push_back(vertexIds);
173 
174  return;
175  }
176 
177  // Add the centroid
178  std::array<double, 3> centroid = {{0., 0., 0.}};
179  for (int k = 0; k < nVertices; ++k) {
180  centroid += m_coordinates[vertexIds[k]];
181  }
182  centroid = centroid / double(nVertices);
183 
184  int centroidId = importVertexCoordinates(std::move(centroid));
185 
186  // Decompose the polygon in triangles
187  //
188  // Each triangle is composed by the two vertices of a side and by the
189  // centroid.
190  ElementType tileType = ElementType::TRIANGLE;
191  int nTileVertices = ReferenceElementInfo::getInfo(tileType).nVertices;
192  int nSideVertices = ReferenceElementInfo::getInfo(ElementType::LINE).nVertices;
193 
194  int nSides = nVertices;
195  m_types.resize(m_nTiles + nSides, tileType);
196  m_connects.resize(m_nTiles + nSides, std::vector<int>(nTileVertices));
197  for (int i = 0; i < nSides; ++i) {
198  m_nTiles++;
199  for (int k = 0; k < nSideVertices; ++k) {
200  m_connects[m_nTiles - 1][k] = vertexIds[(i + k) % nVertices];
201  }
202  m_connects[m_nTiles - 1][nSideVertices] = centroidId;
203  }
204 }

212 void Element::Tesselation::importPolyhedron(const std::vector<int> &vertexIds, const std::vector<std::vector<int>> &faceVertexIds)
213 {
214  int nFaces = faceVertexIds.size();
215  int nVertices = vertexIds.size();
216 
217  // Generate the tesselation of the surface
218  int nInitialTiles = m_nTiles;
219  for (int i = 0; i < nFaces; ++i) {
220  importPolygon(faceVertexIds[i]);
221  }
222  int nFinalTiles = m_nTiles;
223 
224  // Add the centroid of the element to the tesselation
225  std::array<double, 3> centroid = {{0., 0., 0.}};
226  for (int k = 0; k < nVertices; ++k) {
227  centroid += m_coordinates[vertexIds[k]];
228  }
229  centroid = centroid / double(nVertices);
230 
231  int centroidTesselationId = importVertexCoordinates(std::move(centroid));
232 
233  // Decompose the polyhedron in prisms and pyramids.
234  //
235  // Each prism/pyramid has a tile of the surface tesselation as the
236  // base and the centroid of the element as the apex. Since we have
237  // already generated the surface tesselation, we can "convert" the
238  // two-dimensional tiles of that tesselation in three-dimensional
239  // tiles to obtain the volume tesselation.
240  for (int i = nInitialTiles; i < nFinalTiles; ++i) {
241  // Change the tile type
242  ElementType &tileType = m_types[i];
243  if (tileType == ElementType::TRIANGLE) {
244  tileType = ElementType::TETRA;
245  } else if (tileType == ElementType::QUAD) {
246  tileType = ElementType::PYRAMID;
247  } else {
248  BITPIT_UNREACHABLE("Unsupported tile");
249  throw std::runtime_error ("Unsupported tile");
250  }
251 
252  // Fix the order of the connectivity the match the order of the
253  // three-dimensional element
254  if (tileType == ElementType::TETRA) {
255  std::swap(m_connects[i][0], m_connects[i][2]);
256  } else if (tileType == ElementType::PYRAMID) {
257  std::swap(m_connects[i][1], m_connects[i][3]);
258  }
259 
260  // Add the centroid to the tile connectivity
261  m_connects[i].push_back(centroidTesselationId);
262  }
263 }

746 ConstProxyVector<int> Element::getFaceLocalConnect(const int &face) const
747 {
748  switch (m_type) {
749 
750  case (ElementType::POLYGON):
751  {
752  int connectSize = getConnectSize();
753 
754  int faceConnectSize = getFaceVertexCount(face);
755  std::vector<int> localFaceConnect(faceConnectSize);
756  for (int i = 0; i < faceConnectSize; ++i) {
757  int localVertexId = (face + i) % connectSize;
758 
759  localFaceConnect[i] = localVertexId;
760  }
761 
762  return ConstProxyVector<int>(std::move(localFaceConnect));
763  }
764 
765  case (ElementType::POLYHEDRON):
766  {
767  ElementType faceType = getFaceType(face);
768  ConstProxyVector<long> faceVertexIds = getFaceVertexIds(face);
769  int nFaceVertices = faceVertexIds.size();
770  bool faceHasReferenceInfo = ReferenceElementInfo::hasInfo(faceType);
771 
772  int faceConnectSize = nFaceVertices;
773  int localVertexOffset = 0;
774  if (!faceHasReferenceInfo) {
775  ++faceConnectSize;
776  ++localVertexOffset;
777  }
778 
779  std::vector<int> localFaceConnect(faceConnectSize);
780  if (!faceHasReferenceInfo) {
781  localFaceConnect[0] = nFaceVertices;
782  }
783 
784  // The list of the vertices in sorted, therfore we can search the
785  // vertices using a binary search.
786  ConstProxyVector<long> vertexIds = getVertexIds();
787  for (int k = 0; k < nFaceVertices; ++k) {
788  int vertexId = faceVertexIds[k];
789  auto localVertexIdItr = std::lower_bound(vertexIds.begin(), vertexIds.end(), vertexId);
790  assert(localVertexIdItr != vertexIds.end() && *localVertexIdItr == vertexId);
791  localFaceConnect[localVertexOffset + k] = std::distance(vertexIds.begin(), localVertexIdItr);
792  }
793 
794  return ConstProxyVector<int>(std::move(localFaceConnect));
795  }
796 
797  default:
798  {
799  assert(m_type != ElementType::UNDEFINED);
800 
801  const std::vector<int> &localFaceConnect = getInfo().faceConnect[face];
802 
803  return ConstProxyVector<int>(localFaceConnect.data(), localFaceConnect.size());
804  }
805 
806  }
807 }

815 ConstProxyVector<long> Element::getFaceConnect(int face) const
816 {
817  const long *connectivity = getConnect();
818 
819  switch (m_type) {
820 
821  case (ElementType::POLYGON):
822  {
823  int connectSize = getConnectSize();
824 
825  int facePos = 1 + face;
826  int faceConnectSize = getFaceVertexCount(face);
827  if (facePos + faceConnectSize <= connectSize) {
828  return ConstProxyVector<long>(connectivity + facePos, faceConnectSize);
829  } else {
830  std::vector<long> faceConnect(faceConnectSize);
831  for (int i = 0; i < faceConnectSize; ++i) {
832  int position = facePos + i;
833  if (position >= connectSize) {
834  position = position % connectSize + 1;
835  }
836 
837  faceConnect[i] = connectivity[position];
838  }
839 
840  return ConstProxyVector<long>(std::move(faceConnect));
841  }
842  }
843 
844  case (ElementType::POLYHEDRON):
845  {
846  ElementType faceType = getFaceType(face);
847 
848  int facePos = getFaceStreamPosition(face);
849  int faceConnectSize = connectivity[facePos];
850  int faceConnectBegin = facePos + 1;
851  if (!ReferenceElementInfo::hasInfo(faceType)) {
852  faceConnectSize++;
853  faceConnectBegin--;
854  }
855 
856  return ConstProxyVector<long>(connectivity + faceConnectBegin, faceConnectSize);
857  }
858 
859  default:
860  {
861  assert(m_type != ElementType::UNDEFINED);
862 
863  ConstProxyVector<int> localFaceConnect = getFaceLocalConnect(face);
864  int faceConnectSize = localFaceConnect.size();
865 
866  std::vector<long> faceConnect(faceConnectSize);
867  for (int k = 0; k < faceConnectSize; ++k) {
868  int localVertexId = localFaceConnect[k];
869  long vertexId = connectivity[localVertexId];
870  faceConnect[k] = vertexId;
871  }
872 
873  return ConstProxyVector<long>(std::move(faceConnect));
874  }
875 
876  }
