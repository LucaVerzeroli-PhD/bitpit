570 std::vector<double> RBFKernel::evalRBF( const std::array<double,3> &point)
571 {
572  std::vector<double> values(m_fields, 0.);
573  int i, j;
574  double dist, basis;
575 
576  for( i=0; i<m_nodes; ++i ){
577  if( m_activeNodes[i] ) {
578  dist = calcDist(point, i) / m_supportRadius;
579  basis = evalBasis( dist );
580 
581  for( j=0; j<m_fields; ++j) {
582  values[j] += basis * m_weight[j][i];
583  }
584  }
585  }
586 
587  return values;
588 }

629 int RBFKernel::solve()
630 {
631  if(m_mode == RBFMode::PARAM) {
632  return -1;
633  }
634 
635  int j, k;
636  double dist;
637 
638  int nS = getActiveCount();
639  int nrhs = getDataCount();
640 
641  int lda = nS;
642  int ldb = nS;
643  int info;
644  int ipiv[nS];
645 
646  std::vector<int> activeSet( getActiveSet() );
647 
648  std::vector<double> A(lda * nS);
649  std::vector<double> b(ldb * nrhs);
650 
651  k=0;
652  for( j=0; j<nrhs; ++j) {
653  for( const auto & i : activeSet ) {
654  b[k] = m_value[j][i];
655  ++k;
656  }
657  }
658 
659  k=0;
660  for( const auto &i : activeSet ) {
661  for( const auto &j : activeSet ){
662 
663  dist = calcDist(j,i) / m_supportRadius;
664  A[k] = evalBasis( dist );
665  k++;
666  }
667  }
668 
669  info = LAPACKE_dgesv( LAPACK_COL_MAJOR, nS, nrhs, A.data(), lda, ipiv, b.data(), ldb );
670 
671  if( info > 0 ) {
672  printf( "The diagonal element of the triangular factor of the linear system matrix \n" );
673  printf( "U(%i,%i) is zero, so that matrix is singular;\n", info, info );
674  printf( "the solution could not be computed.\n" );
675  return 1;
676  }
677 
678  m_weight.resize(nrhs);
679 
680  k=0;
681  for( j=0; j<nrhs; ++j) {
682  m_weight[j].resize(m_nodes,0);
683 
684  for( const auto &i : activeSet ) {
685  m_weight[j][i] = b[k];
686  ++k;
687  }
688  }
689 
690  return 0;
691 }

721  }

833 double RBFKernel::evalError( )
834 {
835  if(m_mode == RBFMode::PARAM) {
836  return -1.0;
837  }
838 
839  int i(0), j(0);
840  //int index;
841  double maxError(0), relError, realValue, norm;
842  std::vector<double> reconValues;
843 
844  for( int iX=0; iX< m_nodes; ++iX ) {
845  reconValues = evalRBF( iX );
846 
847  j=0;
848  norm = 0.;
849  relError = 0.;
850  for( auto &val : reconValues ) {
851  realValue = m_value[j][i];
852  relError += std::pow( (val - realValue), 2 );
853  norm += std::pow( realValue, 2 );
854 
855  ++j;
856  }
857 
858  relError = sqrt(relError);
859  m_error[i] = relError;
860 
861  if( relError > maxError ) {
862  maxError = relError;
863  }
864 
865  ++i;
866  }
867 
868  return maxError;
869 }

877 int RBFKernel::solveLSQ()
878 {
879  if(m_mode == RBFMode::PARAM) {
880  return -1;
881  }
882 
883  int i, j, k;
884  double dist;
885 
886  int nR = getActiveCount();
887  int nP = m_nodes;
888  int nrhs = getDataCount();
889 
890  std::vector<int> activeSet( getActiveSet() );
891 
892  int n ,m, lda, ldb, info, rank;
893  double rcond = -1.0;
894  //double rcond = 1.e-4;
895 
896  m = nP;
897  n = nR;
898 
899  lda = m;
900  ldb = std::max(n,m);
901 
902  std::vector<double> A(lda * n);
903  std::vector<double> b(ldb * nrhs);
904  std::vector<double> s(m);
905 
906  for( j=0; j<nrhs; ++j) {
907  for( i=0; i<nP; ++i) {
908  k = j*ldb + i;
909  b[k] = m_value[j][i];
910  }
911  }
912 
913  k=0;
914  for( const auto &j : activeSet ) {
915  for( i=0; i<nP; ++i) {
916  dist = calcDist(j,i) / m_supportRadius;
917  A[k] = evalBasis( dist );
918  k++;
919  }
920  }
921 
922  info = LAPACKE_dgelsd( LAPACK_COL_MAJOR, nP, nR, nrhs, A.data(), lda, b.data(), ldb, s.data(), rcond, &rank );
923 
924  if( info > 0 ) {
925  return 1;
926  }
927 
928  m_weight.resize(nrhs);
929 
930  for( j=0; j<nrhs; ++j) {
931  m_weight[j].clear();
932  m_weight[j].resize(m_nodes,0);
933 
934  k=0;
935  for( const auto &i : activeSet ) {
936  m_weight[j][i] = b[j*ldb+k];
937  ++k;
938  }
939  }
940 
941  return 0;
942 }
