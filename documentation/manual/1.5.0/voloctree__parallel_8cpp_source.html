101 std::vector<adaption::Info> VolOctree::_partitioningPrepare(bool trackPartitioning)
102 {
103  std::vector<adaption::Info> partitioningData;
104  if (trackPartitioning) {
105  int currentRank = getRank();
106  PabloUniform::LoadBalanceRanges loadBalanceRanges = m_tree->evalLoadBalanceRanges((std::vector<double> *) nullptr);
107  for (const auto &entry : loadBalanceRanges.sendRanges) {
108  int receiver = entry.first;
109  if (receiver == currentRank) {
110  continue;
111  }
112 
113  adaption::Type adaptionType;
114  if (loadBalanceRanges.sendAction == PabloUniform::LoadBalanceRanges::ACTION_DELETE) {
115  adaptionType = adaption::TYPE_DELETION;
116  } else {
117  adaptionType = adaption::TYPE_PARTITION_SEND;
118  }
119 
120  partitioningData.emplace_back();
121  adaption::Info &partitioningInfo = partitioningData.back();
122  partitioningInfo.entity = adaption::ENTITY_CELL;
123  partitioningInfo.type = adaptionType;
124  partitioningInfo.rank = receiver;
125 
126  const std::array<uint32_t, 2> &sendRange = entry.second;
127  uint32_t beginTreeId = sendRange[0];
128  uint32_t endTreeId = sendRange[1];
129  partitioningInfo.previous.reserve(endTreeId - beginTreeId);
130  for (uint32_t treeId = beginTreeId; treeId < endTreeId; ++treeId) {
131  OctantInfo octantInfo(treeId, true);
132  partitioningInfo.previous.emplace_back();
133  long &cellId = partitioningInfo.previous.back();
134  cellId = getOctantId(octantInfo);
135  }
136  }
137  }
138 
139  return partitioningData;
140 }
