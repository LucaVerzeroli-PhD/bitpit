116 void SegmentationKernel::setSurface( const SurfUnstructured *surface, double featureAngle){
117 
118  std::vector<std::array<double,3>> vertexNormal ;
119  std::vector<std::array<double,3>> vertexGradient ;
120 
121  m_surface = surface;
122  m_featureAngle = featureAngle;
123 
124  double tol = m_surface->getTol() ;
125  for( const Cell &segment : m_surface->getCells() ){
126  long segmentId = segment.getId() ;
127  int nVertices = segment.getVertexCount() ;
128 
129  vertexNormal.resize(nVertices) ;
130  vertexGradient.resize(nVertices) ;
131 
132  double misalignment = 0. ;
133  for( int i = 0; i < nVertices; ++i ){
134  vertexGradient[i] = m_surface->evalVertexNormal(segmentId, i) ;
135  vertexNormal[i] = m_surface->evalLimitedVertexNormal(segmentId, i, m_featureAngle) ;
136 
137  misalignment += norm2(vertexGradient[i] - vertexNormal[i]) ;
138  }
139 
140  m_vertexGradients.insert({{segmentId, vertexGradient}}) ;
141  if( misalignment >= tol ){
142  m_vertexNormals.insert({{segmentId, vertexNormal}}) ;
143  }
144  }
145 
146  // Initialize search tree
147  m_searchTreeUPtr = std::unique_ptr<SurfaceSkdTree>(new SurfaceSkdTree(surface));
148  m_searchTreeUPtr->build();
149 }

183 void SegmentationKernel::getSegmentInfo( const std::array<double,3> &pointCoords, const long &segmentId, const bool &signd, double &distance, std::array<double,3> &gradient, std::array<double,3> &normal ) const {
184 
185 
186  std::array<double,3> outwards;
187 
188  auto itrNormal = getVertexNormals().find(segmentId) ;
189  auto itrGradient = getVertexGradients().find(segmentId) ;
190  assert( itrGradient != getVertexGradients().end() ) ;
191 
192  const Cell &cell = m_surface->getCell(segmentId) ;
193  ElementType cellType = cell.getType();
194  const long *cellConnect = cell.getConnect();
195 
196  std::array<double,3> projectionCoords;
197 
198  switch (cellType) {
199 
200  case ElementType::VERTEX :
201  {
202  long id = cellConnect[0] ;
203 
204  projectionCoords = m_surface->getVertexCoords(id);
205 
206 
207  normal.fill(0.);
208 
209  break;
210  }
211 
212  case ElementType::LINE:
213  {
214  long id0 = cellConnect[0] ;
215  long id1 = cellConnect[1] ;
216  std::array<double,2> lambda ;
217 
218  projectionCoords = CGElem::projectPointSegment( pointCoords, m_surface->getVertexCoords(id0), m_surface->getVertexCoords(id1), lambda);
219  outwards = lambda[0] *itrGradient->second[0] ;
220  outwards += lambda[1] *itrGradient->second[1] ;
221  outwards /= norm2(outwards) ;
222 
223  if( itrNormal != getVertexNormals().end() ){
224  normal = lambda[0] *itrNormal->second[0] ;
225  normal += lambda[1] *itrNormal->second[1] ;
226  normal /= norm2(normal) ;
227 
228  } else {
229  normal = outwards;
230 
231  }
232 
233  break;
234  }
235 
236  case ElementType::TRIANGLE:
237  {
238  long id0 = cellConnect[0] ;
239  long id1 = cellConnect[1] ;
240  long id2 = cellConnect[2] ;
241 
242  std::array<double,3> lambda ;
243 
244  projectionCoords = CGElem::projectPointTriangle( pointCoords, m_surface->getVertexCoords(id0), m_surface->getVertexCoords(id1), m_surface->getVertexCoords(id2), lambda );
245  outwards = lambda[0] *itrGradient->second[0] ;
246  outwards += lambda[1] *itrGradient->second[1] ;
247  outwards += lambda[2] *itrGradient->second[2] ;
248  outwards /= norm2(outwards);
249 
250 
251  if( itrNormal != getVertexNormals().end() ){
252  normal = lambda[0] *itrNormal->second[0] ;
253  normal += lambda[1] *itrNormal->second[1] ;
254  normal += lambda[2] *itrNormal->second[2] ;
255  normal /= norm2(normal) ;
256 
257  } else {
258  normal = outwards;
259 
260  }
261 
262  break;
263  }
264 
265  default:
266  {
267  std::runtime_error ("Type of cell not supported.");
268  break;
269  }
270 
271  }
272 
273  gradient = pointCoords-projectionCoords;
274  distance = norm2(gradient);
275  gradient /= distance;
276 
277  // the sign is computed by determining the side of point p
278  // with respect to the normal plane
279  double s = sign( dotProduct(gradient, outwards) );
280 
281  // if p lies on the normal plane (s=0), but the distance is finite the sign must be evaluated
282  // considering the curvature of the surface. Anyhow this situation is not crucial because
283  // there should exists another element with smaller distance. The signed is put aribtrarly
284  // to positive
285  if(utils::DoubleFloatingEqual()(s,0.) && distance>0){
286  s = 1.;
287  }
288 
289  // If signed distance are computed, the distance value and gradient
290  // need to be changed accordingly. If unsigned distance are computed
291  // the orientation of the suraface normal is discarded and in order
292  // to agnostic with repect the two sides of the surface
293  distance *= ( signd *s + (!signd) *1.);
294  gradient *= ( signd *s + (!signd) *1.);
295  normal *= ( signd *1. + (!signd) *s );
296 
297  return;
298 
299 }

534 bool LevelSetSegmentation::seedNarrowBand( LevelSetCartesian *visitee, std::vector<std::array<double,3>> &VS, double searchRadius, std::vector<long> &I){
535 
536  VolCartesian &mesh = *(static_cast<VolCartesian*>(visitee->getMesh()));
537 
538  bool found(false) ;
539  int dim( mesh.getDimension() ) ;
540  std::array<double,3> B0, B1;
541  std::vector<std::array<double,3>> VP ;
542 
543  mesh.getBoundingBox(B0, B1) ;
544 
545  for( int i=0; i<dim; ++i){
546  B0[i] -= searchRadius;
547  B1[i] += searchRadius;
548  }
549 
550  I.clear() ;
551 
552  for( const auto &P : VS){
553  if( CGElem::intersectPointBox( P, B0, B1, dim ) ) {
554  I.push_back( mesh.locateClosestCell(P) );
555  found = true ;
556  }
557  }
558 
559  if( !found && CGElem::intersectBoxPolygon( B0, B1, VS, false, true, true, VP, dim ) ) {
560  for( const auto &P : VP){
561  I.push_back( mesh.locateClosestCell(P) );
562  found = true ;
563  }
564  }
565 
566  return found ;
567 }

640 void LevelSetSegmentation::computeLSInNarrowBand( LevelSetCartesian *visitee, bool signd){
641 
642  VolCartesian &mesh = *(visitee->getCartesianMesh() ) ;
643  double searchRadius = m_narrowBand;
644 
645  if(searchRadius<0.){
646  for( int d=0; d < mesh.getDimension(); ++d){
647  searchRadius = std::max( searchRadius, mesh.getSpacing(d) ) ;
648  }
649  }
650 
651  std::vector<std::array<double,3>> VS(3);
652 
653  const SurfUnstructured &m_surface = m_segmentation->getSurface();
654 
655  std::vector<long> stack, temp, neighs, flag( mesh.getCellCount(), -1);
656 
657  std::vector< std::array<double,3> > cloud ;
658  std::vector<double> cloudDistance;
659 
660  double distance;
661  std::array<double,3> gradient, normal;
662 
663  stack.reserve(128) ;
664  temp.reserve(128) ;
665 
666  log::cout() << " Compute levelset on cartesian mesh" << std::endl;
667 
668  for (const Cell &segment : m_surface.getCells()) {
669  long segmentId = segment.getId();
670 
671  // compute initial seeds, ie the cells where the vertices
672  // of the surface element fall in and add them to stack
673  m_segmentation->getSegmentVertexCoords( segmentId, &VS ) ;
674  seedNarrowBand( visitee, VS, searchRadius, stack );
675 
676  ElementType segmentType = segment.getType();
677 
678 
679  // propagate from seed
680  size_t stackSize = stack.size();
681  while (stackSize > 0) {
682 
683  // put the cell centroids of the stack into a vector
684  // and calculate the distances to the cloud
685  cloud.resize(stackSize) ;
686  cloudDistance.resize(stackSize);
687 
688  for( size_t k = 0; k < stackSize; ++k) {
689  long cell = stack[k];
690  cloud[k] = visitee->computeCellCentroid(cell) ;
691  }
692 
693  switch (segmentType) {
694 
695  case ElementType::VERTEX :
696  {
697  for( size_t k=0; k<stackSize; ++k){
698  cloudDistance[k] = norm2( cloud[k]-VS[0]);
699  }
700  break;
701  }
702 
703  case ElementType::LINE:
704  {
705  for( size_t k=0; k<stackSize; ++k){
706  cloudDistance[k] = CGElem::distancePointSegment( cloud[k], VS[0], VS[1]);
707  }
708  break;
709  }
710 
711  case ElementType::TRIANGLE:
712  {
713  cloudDistance = CGElem::distanceCloudTriangle( cloud, VS[0], VS[1], VS[2]);
714  break;
715  }
716 
717  default:
718  {
719  std::runtime_error ("Type of cell not supported.");
720  break;
721  }
722  }
723 
724  // check each cell of cloud individually
725  for( size_t k = 0; k < stackSize; ++k) {
726 
727  long &cellId = stack[k];
728  double &cellDistance = cloudDistance[k];
729 
730  // consider only cells within the search radius
731  if ( cellDistance <= searchRadius ) {
732 
733  PiercedVector<LevelSetInfo>::iterator lsInfoItr = m_ls.find(cellId) ;
734  if( lsInfoItr == m_ls.end() ){
735  lsInfoItr = m_ls.emplace(cellId) ;
736  }
737 
738  // check if the computed distance is the closest distance
739  if( cellDistance < std::abs(lsInfoItr->value) ){
740 
741  // compute all necessary information and store them
742  m_segmentation->getSegmentInfo(cloud[k], segmentId, signd, distance, gradient, normal);
743 
744  lsInfoItr->value = distance;
745  lsInfoItr->gradient = gradient;
746 
747  auto infoItr = m_surfaceInfo.find(cellId) ;
748  if( infoItr == m_surfaceInfo.end() ){
749  infoItr = m_surfaceInfo.emplace(cellId) ;
750  }
751  infoItr->support = segmentId;
752  infoItr->normal = normal;
753  }
754 
755 
756  // the new stack is composed of all neighbours
757  // of the old stack. Attention must be paid in
758  // order not to evaluate the same cell twice
759  neighs.clear();
760  mesh.findCellFaceNeighs(cellId, &neighs) ;
761  for( const auto & neigh : neighs){
762  if( flag[neigh] != segmentId) {
763  temp.push_back( neigh) ;
764  flag[neigh] = segmentId ;
765  }
766  }
767 
768  } //end if distance
769  }
770 
771  stack.clear() ;
772  stack.swap( temp ) ;
773  stackSize = stack.size() ;
774 
775  } //end while
776  }
777 
778  return;
779 }

793 void LevelSetSegmentation::computeLSInNarrowBand( LevelSetOctree *visitee, bool signd){
794 
795  VolumeKernel &mesh = *(visitee->getMesh()) ;
796 
797  bool adaptiveSearch(m_narrowBand<0);
798  double searchRadius = m_narrowBand;
799  double factor = 0.5 *sqrt( (double) mesh.getDimension() );
800 
801  long segmentId;
802  double distance;
803  std::array<double,3> gradient, normal;
804  std::array<double,3> centroid, root;
805 
806  std::unordered_set<long> intersects;
807 
808  for( const Cell &cell : mesh.getCells() ){
809 
810  long cellId = cell.getId();
811 
812  centroid = visitee->computeCellCentroid(cellId);
813 
814  if(adaptiveSearch){
815  double cellSize = mesh.evalCellSize(cellId);
816  searchRadius = factor *cellSize;
817  }
818 
819  m_segmentation->m_searchTreeUPtr->findPointClosestCell(centroid, searchRadius, &segmentId, &distance);
820 
821  if(segmentId>=0){
822 
823  m_segmentation->getSegmentInfo(centroid, segmentId, signd, distance, gradient, normal);
824 
825  PiercedVector<LevelSetInfo>::iterator lsInfoItr = m_ls.find(cellId);
826  lsInfoItr = m_ls.emplace(cellId) ;
827  lsInfoItr->value = distance;
828  lsInfoItr->gradient = gradient;
829 
830  PiercedVector<SurfaceInfo>::iterator infoItr = m_surfaceInfo.emplace(cellId);
831  infoItr->support = segmentId;
832  infoItr->normal = normal;
833 
834  if(adaptiveSearch){
835  intersects.insert(cellId);
836  }
837 
838  }
839 
840  }
841 
842  if(!adaptiveSearch){
843  return;
844  }
845 
846  for( const long &cellId : intersects){
847 
848  Cell const &cell = mesh.getCell(cellId);
849 
850  centroid = visitee->computeCellCentroid(cellId);
851  root = computeProjectionPoint(cellId);
852 
853  const long *neighbours = cell.getAdjacencies() ;
854  int nNeighbours = cell.getAdjacencyCount() ;
855  for (int n = 0; n < nNeighbours; ++n) {
856  long neighId = neighbours[n];
857 
858  if (neighId < 0) {
859  continue;
860  }
861 
862  // skip if neigh cell has already been processed
863  // either because it is intersected by surface or
864  // because it is a neigh to a previous intersect
865  if( m_ls.count(neighId) != 0 ){
866  continue;
867  }
868 
869  centroid = visitee->computeCellCentroid(neighId);
870  searchRadius = 1.05 *norm2(centroid-root);
871  m_segmentation->m_searchTreeUPtr->findPointClosestCell(centroid, searchRadius, &segmentId, &distance);
872 
873  if(segmentId>=0){
874 
875  m_segmentation->getSegmentInfo(centroid, segmentId, signd, distance, gradient, normal);
876 
877  PiercedVector<LevelSetInfo>::iterator lsInfoItr = m_ls.emplace(neighId) ;
878  lsInfoItr->value = distance;
879  lsInfoItr->gradient = gradient;
880 
881 
882  PiercedVector<SurfaceInfo>::iterator infoItr = m_surfaceInfo.emplace(neighId);
883  infoItr->support = segmentId;
884  infoItr->normal = normal;
885 
886  } else {
887  assert(false && "Should not pass here");
888  }
889 
890  }
891  }
892 }

907 void LevelSetSegmentation::updateLSInNarrowBand( LevelSetOctree *visitee, const std::vector<adaption::Info> &mapper, bool signd){
908 
909  VolumeKernel &mesh = *(visitee->getMesh()) ;
910 
911  bool adaptiveSearch(m_narrowBand<0);
912  double searchRadius = m_narrowBand;
913  double factor = 0.5 *sqrt( (double) mesh.getDimension() );
914 
915  long segmentId;
916  std::array<double,3> centroid, root;
