217 double LevelSetBoolean::getMinSurfaceFeatureSize() const {
218 
219  bool minimumValid = false;
220  double minimumSize = levelSetDefaults::SIZE;
221  for( const auto & object : m_objPtr ){
222  double objectMinimumSize = object->getMinSurfaceFeatureSize();
223  if (objectMinimumSize < 0) {
224  continue;
225  }
226 
227  minimumValid = true;
228  minimumSize = std::min(objectMinimumSize, minimumSize);
229  }
230 
231  if (!minimumValid) {
232  minimumSize = - levelSetDefaults::SIZE;
233  }
234 
235  return minimumSize;

253 /*
254  * Determines the relevant object which determines the levelset value in the cell
255  * Taken from http://www.iue.tuwien.ac.at/phd/ertl/node57.html
256  * @param[in] id cell index
257  * @param[in,out] factor if is not null, on output it will contain the multiplier
258  * of the levelset function of the competent primary object according the boolean
259  * operation. In particular, for substractions the relevant levelset could be
260  * -1.0*levelset_primary_object.
261  * @return pointer to competent LevelSetObject
262  */
263 LevelSetObject* LevelSetBoolean::getCompetentObject( const long &id, double *factor) const{
264 
265  if(m_objPtr.size()==0){
266  return nullptr;
267  }
268 
269  double result, second;
270  LevelSetObject *resPtr, *secPtr;
271 
272  result = m_objPtr[0]->getLS(id);
273  resPtr = m_objPtr[0];
274 
275  for( size_t n=1; n<m_objPtr.size(); ++n){
276  second = m_objPtr[n]->getLS(id) ;
277  secPtr = m_objPtr[n];
278 
279  if( getBooleanOperation() == LevelSetBooleanOperation::UNION){
280  if(result>second) {
281  result = second;
282  resPtr = secPtr;
283  }
284 
285  } else if ( getBooleanOperation() == LevelSetBooleanOperation::INTERSECTION){
286  if(result<second) {
287  result = second;
288  resPtr = secPtr;
289  }
290 
291  } else if ( getBooleanOperation() == LevelSetBooleanOperation::SUBTRACTION){
292  if(result<-1.*second) {
293  result = -1.*second;
294  resPtr = secPtr;
295  }
296  }
297  }
298 
299  if(factor){
300  *factor = (utils::DoubleFloatingEqual()(resPtr->getLS(id),result)) ? 1. : -1.;
301  }
302 
303  return resPtr;
304 }

312 LevelSetInfo LevelSetBoolean::booleanOperation(const long &id) const{
313 
314  if(m_objPtr.empty()){
315  return LevelSetInfo();
316  }
317 
318  LevelSetInfo result = m_objPtr[0]->getLevelSetInfo(id);
319  for( size_t n=1; n<m_objPtr.size(); ++n){
320  double value = m_objPtr[n]->getLS(id) ;
321 
322  if( getBooleanOperation() == LevelSetBooleanOperation::UNION){
323  if(result.value>value) {
324  result = m_objPtr[n]->getLevelSetInfo(id);
325  }
326 
327  } else if ( getBooleanOperation() == LevelSetBooleanOperation::INTERSECTION){
328  if(result.value<value) {
329  result = m_objPtr[n]->getLevelSetInfo(id);
330  }
331 
332  } else if ( getBooleanOperation() == LevelSetBooleanOperation::SUBTRACTION){
333  if(result.value<-value) {
334  result = m_objPtr[n]->getLevelSetInfo(id);
335  result.value *= -1.;
336  result.gradient *= -1.;
337  }
338  }
339  }
340 
341  return result;
342 }
