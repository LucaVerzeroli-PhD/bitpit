255 void DataCommunicator::discoverSends()
256 {
257  // Cancel current sends
258  clearAllSends();
259 
260  // Send the data sizes with a synchronous send
261  int nRecvs = getRecvCount();
262 
263  std::vector<long> discoverSizes(nRecvs, 0);
264  std::vector<MPI_Request> discoverRequests(nRecvs, MPI_REQUEST_NULL);
265  for (int i = 0; i < nRecvs; ++i) {
266  int rank = m_recvRanks[i];
267  RecvBuffer &buffer = m_recvBuffers[i];
268  discoverSizes[i] = buffer.getSize();
269  MPI_Issend(discoverSizes.data() + i, 1, MPI_LONG, rank, m_discoverTag, m_communicator, discoverRequests.data() + i);
270  }
271 
272  // Receive the data sizes and set the sends
273  MPI_Request exchangeCompletedRequest = MPI_REQUEST_NULL;
274  while (true) {
275  // If there are messagea available receive them and set the sends
276  int messageAvailable = 1;
277  while (messageAvailable) {
278  MPI_Status status;
279  MPI_Iprobe(MPI_ANY_SOURCE, m_discoverTag, m_communicator, &messageAvailable, &status);
280  if (messageAvailable) {
281  long dataSize;
282  MPI_Recv(&dataSize, 1, MPI_LONG, status.MPI_SOURCE, m_discoverTag, m_communicator, MPI_STATUS_IGNORE);
283  setSend(status.MPI_SOURCE, dataSize);
284  }
285  }
286 
287  // If all the sends are complete notify it
288  if (exchangeCompletedRequest == MPI_REQUEST_NULL) {
289  int discoverSendsCompleted;
290  MPI_Testall(discoverRequests.size(), discoverRequests.data(), &discoverSendsCompleted, MPI_STATUSES_IGNORE);
291  if (discoverSendsCompleted) {
292  MPI_Ibarrier(m_communicator, &exchangeCompletedRequest);
293  }
294  }
295 
296  // If all sends are completed, check if also the other processes have
297  // completed the sends. Sice these are synchronous sends, they will
298  // be makred as completed only when the corresponding receive has
299  // completed. When all processes have completed the send/recevies
300  // all sizes have been exchanged.
301  if (exchangeCompletedRequest != MPI_REQUEST_NULL) {
302  int exchangeCompleted = 0;
303  MPI_Test(&exchangeCompletedRequest, &exchangeCompleted, MPI_STATUS_IGNORE);
304  if (exchangeCompleted) {
305  break;
306  }
307  }
308  }
309 }

375 void DataCommunicator::clearSend(int rank)
376 {
377  if (m_sendIds.count(rank) == 0) {
378  return;
379  }
380 
381  // Cancel waiting send
382  cancelSend(rank);
383 
384  // Remove the send associated to the specified rank
385  int id = m_sendIds[rank];
386  m_sendIds.erase(rank);
387  for (auto &entry : m_sendIds) {
388  if (entry.second > id) {
389  entry.second--;
390  }
391  }
392 
393  m_sendRanks.erase(m_sendRanks.begin() + id);
394  m_sendRequests.erase(m_sendRequests.begin() + id);
395  m_sendBuffers.erase(m_sendBuffers.begin() + id);
396 }

980 void DataCommunicator::cancelAllSends(bool synchronous)
981 {
982  // If no synchronization is needed just call the standard overload
983  if (!synchronous) {
984  cancelAllSends();
985  return;
986  }
987 
988  // Start auxiliary receives for handling synchronization
989  //
990  // We will receive a notification from all the processes for which a
991  // receive has been set. The notification will contain the status of
992  // the send request on that processor that matches the receive of this
993  // rank. A status equal to zero means that the send has been successfully
994  // canceled.
995  int nRecvs = getRecvCount();
996  std::vector<MPI_Request> recvNotificationRequests(nRecvs);
997 
998  std::vector<int> remoteSendStatus(nRecvs, 1);
999  for (int i = 0; i < nRecvs; ++i) {
1000  int rank = m_recvRanks[i];
1001  MPI_Irecv(remoteSendStatus.data() + i, 1, MPI_INT, rank, m_discoverTag, m_communicator, &recvNotificationRequests[i]);
1002  }
1003 
1004  // Cancel all the sends
1005  //
1006  // We will send a notification to all processes for wich a send has been
1007  // set. The notification will contain the status of the send request. A
1008  // status equal to zero means that the send request has been successfully
1009  // canceled.
1010  int nSends = getSendCount();
1011  std::vector<MPI_Request> sendNotificationRequests(nSends);
1012  for (int i = 0; i < nSends; ++i) {
1013  int rank = m_sendRanks[i];
1014 
1015  // Cancel the send
1016  cancelSend(rank);
1017 
1018  // Notify the other process that the send has been canceled
1019  int sendStatus = 0;
1020  MPI_Isend(&sendStatus, 1, MPI_INT, rank, m_discoverTag, m_communicator, &sendNotificationRequests[i]);
1021  }
1022 
1023  // Synchronize with other processes
1024  //
1025  // Wait until all processes involved in data exchanging cancel their
1026  // send requests.
1027  std::size_t nMissingNotifications = recvNotificationRequests.size();
1028  while (nMissingNotifications > 0) {
1029  int i;
1030  MPI_Status status;
1031  MPI_Waitany(recvNotificationRequests.size(), recvNotificationRequests.data(), &i, &status);
1032  if (remoteSendStatus[i] == 1) {

1060 void DataCommunicator::cancelAllRecvs(bool synchronous)
1061 {
1062  // If no synchronization is needed just call the standard overload
1063  if (!synchronous) {
1064  cancelAllRecvs();
1065  return;
1066  }
1067 
1068  // Start auxiliary receives for handling synchronization
1069  //
1070  // We will receive a notification from all the processes for which a
1071  // send has been set. The notification will contain the status of the
1072  // receive request on that processor that matches the send of this rank.
1073  // A status equal to zero means that the receive has been successfully
1074  // canceled.
1075  int nSends = getSendCount();
1076  std::vector<MPI_Request> recvNotificationRequests(nSends);
1077 
1078  std::vector<int> remoteReceiveStatus(nSends, 1);
1079  for (int i = 0; i < nSends; ++i) {
1080  int rank = m_sendRanks[i];
1081  MPI_Irecv(remoteReceiveStatus.data() + i, 1, MPI_INT, rank, m_discoverTag, m_communicator, &recvNotificationRequests[i]);
1082  }
1083 
1084  // Cancel all the receives
1085  //
1086  // We will send a notification to all processes for wich a receive has been
1087  // set. The notification will contain the status of the receive request. A
1088  // status equal to zero means that the receive request has been successfully
1089  // canceled.
1090  int nRecvs = getRecvCount();
1091  std::vector<MPI_Request> sendNotificationRequests(nRecvs);
1092  for (int i = 0; i < nRecvs; ++i) {
1093  int rank = m_recvRanks[i];
1094 
1095  // Cancel the send
1096  cancelRecv(rank);
1097 
1098  // Notify other process that the received has been canceled
1099  int receiveStatus = 0;
1100  MPI_Isend(&receiveStatus, 1, MPI_INT, rank, m_discoverTag, m_communicator, &sendNotificationRequests[i]);
1101  }
1102 
1103  // Synchronize with other processes
1104  //
1105  // Wait until all processes involved in data exchanging cancel their
1106  // receive requests.
1107  std::size_t nMissingNotifications = recvNotificationRequests.size();
1108  while (nMissingNotifications > 0) {
1109  int i;
1110  MPI_Waitany(recvNotificationRequests.size(), recvNotificationRequests.data(), &i, MPI_STATUS_IGNORE);
1111  if (remoteReceiveStatus[i] == 1) {
1112  log::cout() << "Unable to properly cancel the receives.";
1113  MPI_Abort(m_communicator, 1);
1114  }
1115 
1116  --nMissingNotifications;
1117  }
1118 
1119  // Wait until the notifications have been sent
1120  MPI_Waitall(sendNotificationRequests.size(), sendNotificationRequests.data(), MPI_STATUS_IGNORE);
1121 
1122 #if BITPIT_ENABLE_DEBUG
1123  // Wait until all processes cancel the receives
1124  MPI_Barrier(m_communicator);
1125 #endif
1126 }
