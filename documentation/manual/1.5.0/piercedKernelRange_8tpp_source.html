99 template<typename id_t>
100 std::size_t PiercedKernelRange<id_t>::evalSize() const
101 {
102  const PiercedKernel<id_t> &kernel = m_cbegin.getKernel();
103 
104  std::size_t flatBegin = kernel.evalFlatIndex(m_cbegin.getId());
105 
106  std::size_t flatEnd;
107  if (m_cend.getPos() <= kernel.getLastUsedPos()) {
108  flatEnd = kernel.evalFlatIndex(m_cend.getId()) + 1;
109  } else {
110  flatEnd = kernel.size();
111  }
112 
113  std::size_t size = (flatEnd - flatBegin);
114 
115  return size;
116 }
