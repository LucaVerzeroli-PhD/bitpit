40 inline uint64_t splitBy3(uint32_t a)
41 {
42  uint64_t x = a & 0x1fffff; // we only look at the first 21 bits
43  x = (x | x << 32) & 0x1f00000000ffff; // shift left 32 bits, OR with self, and 00011111000000000000000000000000000000001111111111111111
44  x = (x | x << 16) & 0x1f0000ff0000ff; // shift left 32 bits, OR with self, and 00011111000000000000000011111111000000000000000011111111
45  x = (x | x << 8) & 0x100f00f00f00f00f; // shift left 32 bits, OR with self, and 0001000000001111000000001111000000001111000000001111000000000000
46  x = (x | x << 4) & 0x10c30c30c30c30c3; // shift left 32 bits, OR with self, and 0001000011000011000011000011000011000011000011000011000100000000
47  x = (x | x << 2) & 0x1249249249249249;
48 
49  return x;
50 }

60 inline uint64_t splitBy2(uint32_t a)
61 {
62  uint64_t x = a;
63  x = (x | x << 16) & 0xFFFF0000FFFF; // shift left 16 bits, OR with self, and 0000000000000000111111111111111100000000000000001111111111111111
64  x = (x | x << 8) & 0xFF00FF00FF00FF; // shift left 8 bits, OR with self, and 0000000011111111000000001111111100000000111111110000000011111111
65  x = (x | x << 4) & 0xF0F0F0F0F0F0F0F; // shift left 4 bits, OR with self, and 0000111100001111000011110000111100001111000011110000111100001111
66  x = (x | x << 2) & 0x3333333333333333; // shift left 2 bits, OR with self, and 0011001100110011001100110011001100110011001100110011001100110011
67  x = (x | x << 1) & 0x5555555555555555; // shift left 1 bits, OR with self, and 0101010101010101010101010101010101010101010101010101010101010101
68 
69  return x;
70 }

83 inline uint64_t computeMorton(uint32_t x, uint32_t y, uint32_t z)
84 {
85  uint64_t morton = splitBy3(x) | splitBy3(y) << 1 | splitBy3(z) << 2;
