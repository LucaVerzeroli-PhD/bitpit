99 bitpit::FlatVector2D<long> Cell::createNeighbourhoodStorage(bool storeNeighbourhood)
100 {

256 void Cell::_initialize(bool interior, bool initializeNeighbourhood, bool storeNeighbourhood)
257 {
258  setInterior(interior);
259 
260  // Neighbourhood
261  if (initializeNeighbourhood) {
262  // To reduce memory fragmentation, destroy both interfaces/adjacencies
263  // before resetting the interfaces/adjacencies
264  int reallocateInterfaces = (m_interfaces.getItemCount() != getFaceCount());
265  if (reallocateInterfaces) {
266  m_interfaces.destroy();
267  }
268 
269  int reallocateAdjacencies = (m_adjacencies.getItemCount() != getFaceCount());
270  if (reallocateAdjacencies) {
271  m_adjacencies.destroy();
272  }
273 
274  // Reset the interfaces/adjacencies
275  resetInterfaces(storeNeighbourhood);
276  resetAdjacencies(storeNeighbourhood);
277  }
278 }

335 void Cell::setInterfaces(std::vector<std::vector<long>> &interfaces)
336 {
337  if (getType() == ElementType::UNDEFINED) {
338  return;
339  }
340 
341  // Initialize the interfaces
342  assert(m_interfaces.size() == getFaceCount());
343  m_interfaces.initialize(interfaces);
344 
345  // Check that there is at least one interfaces for each face. If the face
346  // is not associated to an interface, a dummy interface is added.
347  int nFaces = getFaceCount();
348  for (int i = 0; i < nFaces; ++i) {
349  if (m_interfaces.getItemCount(i) == 0) {
350  m_interfaces.pushBackItem(i, NULL_ID);

373 void Cell::pushInterface(const int &face, const long &interface)
374 {
375  // Do not push an existing interface
376  if (findInterface(face, interface) >= 0) {
377  return;
378  }
379 
380  // If there is only one interface stored for the specified face and
381  // that interface is negative, we need to overwrite the value and
382  // not add another interface.
383  if (m_interfaces.getItemCount(face) == 1) {
384  if (m_interfaces.getItem(face, 0) < 0) {
385  m_interfaces.setItem(face, 0, interface);
386  return;
387  }
388  }
389 
390  // There are multiple adjacency, we need to add the interface.
391  m_interfaces.pushBackItem(face, interface);
392 }

804 void Cell::display(std::ostream &out, unsigned short int indent) const
805 {
806  // ====================================================================== //
807  // VARIABLES DECLARATION //
808  // ====================================================================== //
809 
810  // Local variables
811  std::string t_s(indent, ' ');
812 
813  // Counters
814  int i, j;
815  int nn;
816 
817  // ====================================================================== //
818  // DISPLAY INFOS //
819  // ====================================================================== //
820  if (getType() == ElementType::UNDEFINED) {
821  out << t_s << "cell type: (unknown)" << std::endl;
822  return;
823  }
824 
825  // Scope variables -------------------------------------------------- //
826  int nv = getVertexCount();
827  int nf = getFaceCount();
828 
829  ConstProxyVector<long> cellVertexIds = getVertexIds();
830 
831  // General info ----------------------------------------------------- //
832  out << t_s << "cell type: " << getType() << std::endl;
833  out << t_s << "ID: " << getId() << std::endl;
834  out << t_s << "is ghost: ";
835  if (m_interior) { out << "(false)"; }
836  else { out << "(true)"; }
837  out << std::endl;
838 
839  // Connectivity infos --------------------------------------------------- //
840  out << t_s << "connectivity: [ ";
841  for (i = 0; i < nv-1; ++i) {
842  if (cellVertexIds[i] == Vertex::NULL_ID) out << "n.a. ";
843  else out << cellVertexIds[i] << ", ";
844  } //next i
845  if (cellVertexIds[nv-1] == Vertex::NULL_ID) out << "n.a. ";
846  else out << cellVertexIds[nv-1] << " ]" << std::endl;
847 
848  // neighbors infos ------------------------------------------------------ //
849  if (m_adjacencies.size() > 0) {
850  out << t_s << "neighbors: [ ";
851  for (i = 0; i < nf-1; ++i) {
852  nn = getAdjacencyCount(i);
853  out << "[ ";
854  for (j = 0; j < nn-1; ++j) {
855  if (getAdjacency(i,j) == Element::NULL_ID) out << "n.a. ";
856  else out << getAdjacency(i, j) << ", ";
857  } //next j
858  if (getAdjacency(i, nn-1) == Element::NULL_ID) out << "n.a. ], ";
859  else out << getAdjacency(i, nn-1) << " ], ";
860  } //next i
861  nn = getAdjacencyCount(nf-1);
862  out << "[ ";
863  for (j = 0; j < nn-1; ++j) {
864  if (getAdjacency(nf-1, j) == Element::NULL_ID) out << "n.a. ";
865  else out << getAdjacency(nf-1, j) << ", ";
866  } //next j
867  if (getAdjacency(nf-1, nn-1) == Element::NULL_ID) out << "n.a. ]";
868  else out << getAdjacency(nf-1, nn-1) << " ]";
869  out << " ]" << std::endl;
870  }
871 
872  // interface infos ------------------------------------------------------ //
873  if (m_interfaces.size() > 0) {
874  out << t_s << "interfaces: [ ";
875  for (i = 0; i < nf-1; ++i) {
876  nn = getInterfaceCount(i);
877  out << "[ ";
878  for (j = 0; j < nn-1; ++j) {
879  if (getInterface(i,j) == Element::NULL_ID) out << "n.a. ";
880  else out << getInterface(i, j) << ", ";
881  } //next j
882  if (getInterface(i, nn-1) == Element::NULL_ID) out << "n.a. ], ";
883  else out << getInterface(i, nn-1) << " ], ";
884  } //next i
885  nn = getInterfaceCount(nf-1);
886  out << "[ ";
887  for (j = 0; j < nn-1; ++j) {
888  if (getInterface(nf-1, j) == Element::NULL_ID) out << "n.a. ";

890  } //next j

892  else out << getInterface(nf-1, nn-1) << " ]";
893  out << " ]" << std::endl;
894  }
