556  template<class Impl>
557  void
558  communicate(DataCommInterface<Impl> & userData){
559  //BUILD SEND BUFFERS
560  DataCommunicator communicator(m_comm);
561  size_t fixedDataSize = userData.fixedSize();
562  std::map<int,u32vector >::iterator bitend = m_bordersPerProc.end();
563  std::map<int,u32vector >::iterator bitbegin = m_bordersPerProc.begin();
564  for(std::map<int,u32vector >::iterator bit = bitbegin; bit != bitend; ++bit){
565  const int & key = bit->first;
566  const u32vector & pborders = bit->second;
567  size_t buffSize = 0;
568  size_t nofPbordersPerProc = pborders.size();
569  if(fixedDataSize != 0){
570  buffSize = fixedDataSize*nofPbordersPerProc;
571  }
572  else{
573  for(size_t i = 0; i < nofPbordersPerProc; ++i){
574  buffSize += userData.size(pborders[i]);
575  }
576  }
577  //enlarge buffer to store number of pborders from this proc
578  buffSize += sizeof(size_t);
579  //build buffer for this proc
580  communicator.setSend(key,buffSize);
581  SendBuffer & sendBuffer = communicator.getSendBuffer(key);
582  //store number of pborders from this proc at the begining
583  sendBuffer << nofPbordersPerProc;
584 
585  //WRITE SEND BUFFERS
586  for(size_t j = 0; j < nofPbordersPerProc; ++j){
587  userData.gather(sendBuffer,pborders[j]);
588  }
589  }

591  communicator.discoverRecvs();
592  communicator.startAllRecvs();
593 
594  communicator.startAllSends();
