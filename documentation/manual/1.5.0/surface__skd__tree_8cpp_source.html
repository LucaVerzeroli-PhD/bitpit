139 long SurfaceSkdTree::findPointClosestCell(const std::array<double, 3> &point, double maxDistance,
140  long *id, double *distance) const
141 {
142  // Initialize the cell id
143  *id = Cell::NULL_ID;
144 
145  // Initialize the distance with an estimate
146  //
147  // The real distance will be lesser than or equal to the estimate.
148  std::size_t rootId = 0;
149  const SkdNode &root = m_nodes[rootId];
150  *distance = std::min(root.evalPointMaxDistance(point), maxDistance);
151 
152  // Get a list of candidates nodes
153  //
154  // The list of candidates is a memeber of the class to avoid its
155  // reallocation every time the function is called.
156  m_candidateIds.clear();
157  m_candidateMinDistances.clear();
158 
159  std::vector<std::size_t> nodeStack;
160  nodeStack.push_back(rootId);
161  while (!nodeStack.empty()) {
162  std::size_t nodeId = nodeStack.back();
163  const SkdNode &node = m_nodes[nodeId];
164  nodeStack.pop_back();
165 
166  // Do not consider nodes with a minimum distance greater than
167  // the distance estimate
168  double nodeMinDistance = node.evalPointMinDistance(point);
169  if (nodeMinDistance > *distance) {
170  continue;
171  }
172 
173  // Update the distance estimate
174  //
175  // The real distance will be lesser than or equal to the
176  // estimate.
177  double nodeMaxDistance = node.evalPointMaxDistance(point);
178  *distance = std::min(nodeMaxDistance, *distance);
179 
180  // If the node is a leaf add it to the candidates, otherwise
181  // add its children to the stack.
182  bool isLeaf = true;
183  for (int i = SkdNode::CHILD_BEGIN; i != SkdNode::CHILD_END; ++i) {
184  SkdNode::ChildLocation childLocation = static_cast<SkdNode::ChildLocation>(i);
185  std::size_t childId = node.getChildId(childLocation);
186  if (childId != SkdNode::NULL_ID) {
187  isLeaf = false;
188  nodeStack.push_back(childId);
189  }
190  }
191 
192  if (isLeaf) {
193  m_candidateIds.push_back(nodeId);
194  m_candidateMinDistances.push_back(nodeMinDistance);
195  }
196  }
197 
198  // Process the candidates and find the closest cell
199  long nDistanceEvaluations = 0;
200  for (std::size_t k = 0; k < m_candidateIds.size(); ++k) {
201  // Do not consider nodes with a minimum distance greater than
202  // the distance estimate
203  if (m_candidateMinDistances[k] > *distance) {
204  continue;
205  }
206 
207  // Evaluate the distance
208  std::size_t nodeId = m_candidateIds[k];
209  const SkdNode &node = m_nodes[nodeId];
210 
211  long cellId;
212  double cellDistance;
213  node.findPointClosestCell(point, &cellId, &cellDistance);
214  ++nDistanceEvaluations;
215  if (cellDistance < *distance) {
216  *id = cellId;
217  *distance = cellDistance;
218  }
219  }
220 
221  // If no closest cell was found set the distance to the maximum
222  // representable distance.
223  if (*id == Cell::NULL_ID) {
224  *distance = std::numeric_limits<double>::max();
225  }
226 
227  return nDistanceEvaluations;
228 }
